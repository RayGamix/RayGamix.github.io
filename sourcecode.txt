<script>
/* ---------------- GLOBALS ---------------- */
let currentUser = null;
let userTokens = "N/A";
let ugcData = [];
let isPlaying = false;

const Server = {
URL: "https://raygamix.github.io" // your actual server with /Games/list.json
};

/* ---------------- DROPDOWN ---------------- */
function toggleDropdown(id){
document.querySelectorAll('.dropdown-content').forEach(d=>{ if(d.id!==id)d.classList.remove('show'); });
document.getElementById(id).classList.toggle('show');
}

window.onclick=function(event){
if(!event.target.closest(".nav-icon") && !event.target.closest(".dropdown-content")){
document.querySelectorAll('.dropdown-content').forEach(d=>d.classList.remove('show'));
}
};

/* ---------------- SEARCH ---------------- */
document.getElementById("searchInput").addEventListener("input",()=>{
const value=document.getElementById("searchInput").value.toLowerCase();
renderGrid(value);
});

/* ---------------- FETCH & RENDER ---------------- */
async function fetchGames(){
try{
const url=`${Server.URL}/Games/list.json?_=${Date.now()}`; // always fresh
const res=await fetch(url);
ugcData=await res.json();
}catch(e){
// fallback: only 1 demo game
ugcData=[{id:"demo",name:"RayGamix Demo Game",creator:"Raybod"}];
}
renderGrid();
}

function renderGrid(search=""){
const grid=document.getElementById("grid");
grid.innerHTML="";
ugcData.filter(g=>g.name.toLowerCase().includes(search))
.forEach(g=>{
grid.innerHTML+=`
<div class="card">
<div class="thumb"></div>
<div class="card-body">
<div class="card-info">
<div class="card-title notranslate">${g.name}</div>
<div class="card-sub">Creator: <span class="notranslate">${g.creator}</span></div>
</div>
<button class="play" onclick="joinGame('${g.id}')">Play</button>
</div>
</div>`;
});
}

/* ---------------- GAME PLAYER ---------------- */
function joinGame(gameId){
isPlaying=true;
document.getElementById("grid").style.display="none";
document.getElementById("backButtonContainer").style.display="block";
const gameUrl=(Server.URL==="N/A")?"https://raygamix.github.io/demogame.html":`${Server.URL}/Games/${gameId}/index.html`;
const iframe=document.createElement("iframe");
iframe.id="gameFrame";
iframe.src=gameUrl;
iframe.style.width="100%";
iframe.style.height="78vh";
iframe.style.border="none";
iframe.style.borderRadius="12px";
iframe.style.background="#000";
document.getElementById("main").appendChild(iframe);
}

function exitGame(){
isPlaying=false;
const frame=document.getElementById("gameFrame");
if(frame) frame.remove();
document.getElementById("grid").style.display="grid";
document.getElementById("backButtonContainer").style.display="none";
fetchGames();
}

/* ---------------- USER ---------------- */
async function signup(){
const name=prompt("Choose a username:"); if(!name) return;
const pass=prompt("Choose a password:");
currentUser=name; userTokens="N/A"; updateUser();
}

async function login(){
const name=prompt("Username:"); const pass=prompt("Password:");
currentUser=name||"Guest"; userTokens="N/A"; updateUser();
}

function logout(){currentUser=null; userTokens="N/A"; updateUser();}

function updateUser(){
document.getElementById("userDisplay").innerText=currentUser||"";
document.getElementById("stats").innerText="Tokens: "+userTokens;
}

/* ---------------- COLOR ---------------- */
function changeColor(){
const hex=document.getElementById("colorHex").value;
if(!hex) return;
document.documentElement.style.setProperty('--bg',hex);
document.querySelectorAll('.dropdown-content').forEach(d=>d.classList.remove('show'));
}

/* ---------------- AVATAR ---------------- */
function toggleAvatarPanel(){
const panel=document.getElementById("avatarPanel");
panel.style.display=panel.style.display==="flex"?"none":"flex";
}

/* ---------------- START ---------------- */
setInterval(fetchGames,5000); // fetch every 5 seconds
fetchGames();
updateUser();

/* ---------------- GOOGLE TRANSLATE ---------------- */
function googleTranslateElementInit(){
new google.translate.TranslateElement({pageLanguage:'en',includedLanguages:'en,es,fr,de,it,pt,ja,ko,zh-CN,ar',autoDisplay:false},'google_translate_element');
}

function changeLanguage(langCode){
const select=document.querySelector('.goog-te-combo');
if(select){select.value=langCode; select.dispatchEvent(new Event('change'));}
}
